236  template<typename ArrayType>
237  inline std::shared_ptr<FootstepState> makeStateFromMidpose(const ArrayType & midpose, Foot foot) const
238  {
239  // the offset for the nominal foot separation from right to left
240  // rotate the 2D vector (0; nominal_foot_separation / 2.0) by theta
241  double cx_separation_half_offset = -1 * std::sin(midpose[2]) * config_->nominal_foot_separation / 2.0;
242  double cy_separation_half_offset = std::cos(midpose[2]) * config_->nominal_foot_separation / 2.0;
243 
244  int offset_sign = (foot == Foot::LEFT ? 1 : -1);
245  return std::make_shared<FootstepState>(contToDiscXy(midpose[0] + offset_sign * cx_separation_half_offset),
246  contToDiscXy(midpose[1] + offset_sign * cy_separation_half_offset),
247  contToDiscTheta(midpose[2]), foot);
248  }
